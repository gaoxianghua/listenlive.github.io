Ubuntu手动LNMP环境搭建

时间： 2021-11-10   |   分类： 技术     |   阅读： 385 字 ~1分钟

安装Nginx

    sudo apt install nginx

安装MySQL

    sudo apt install mysql-server-5.7

安装PHP

    sudo apt install php7.2
    sudo apt install php-fpm7.2

安装PHP扩展

    sudo apt install php7.2-xml php7.2-mysql php7.2-curl

设置MySQL用户名密码

    #查看初始用户名和密码
    cd /etc/mysql
    sudo cat debian.cnf
    #使用初始用户名密码登录mysql
    mysql -udebian-sys-maint -pGIOyPu42YoZEwRTe
    #成功登录后
    use mysql;
    update mysql.user set authentication_string=password('yourPwd') where user='root' and Host ='localhost';
    update user set plugin="mysql_native_password";
    flush privileges;
    quit;

配置Nginx多域名访问

进入/etc/nginx/sites-available安装目录

    cd /etc/nginx/sites-available

删除默认default默认配置

    sudo rm -rf default

新建配置

    sudo vi mysite.com

    server {
        listen 80;
        listen 443 ssl http2;
        server_name .tp5.test;
        root /www/mysite.com/public;
    
        index index.html index.htm index.php;
    
        charset utf-8;
        client_max_body_size 100M;
    
        
    
        location / {
            if (!-e $request_filename) {
                rewrite  ^(.*)$  /index.php?s=/$1  last;
                break;
            }
            
        }
    
        
    
        location = /favicon.ico { access_log off; log_not_found off; }
        location = /robots.txt  { access_log off; log_not_found off; }
    
        access_log off;
        error_log  /var/log/nginx/tp5.test-error.log error;
    
        sendfile off;
    
        location ~ \.php$ {
            fastcgi_split_path_info ^(.+\.php)(/.+)$;
            fastcgi_pass unix:/var/run/php/php7.2-fpm.sock;
            fastcgi_index index.php;
            include fastcgi_params;
            fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
            
    
            fastcgi_intercept_errors off;
            fastcgi_buffer_size 16k;
            fastcgi_buffers 4 16k;
            fastcgi_connect_timeout 300;
            fastcgi_send_timeout 300;
            fastcgi_read_timeout 300;
        }
    
        location ~ /\.ht {
            deny all;
        }
    
        ssl_certificate     /etc/ssl/certs/mysite.com.crt;
        ssl_certificate_key /etc/ssl/certs/mysite.com.key;
    }

进入/etc/nginx/sites-enabled安装目录

    #删除默认配置的软连接
    sudo rm -rf default
    #设置新的软连接
    sudo ln -s ../sites-available/mysite.com ./

检查配置文件并重启Nginx

    sudo nginx -t
    sudo nginx -s reload
